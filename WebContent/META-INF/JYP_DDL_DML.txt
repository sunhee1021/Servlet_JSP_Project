
/* 회원테이블 */
CREATE TABLE Member (
	EMAIL NVARCHAR2(30) NOT NULL, /* 이메일 */
	PASSWORD NVARCHAR2(20) NOT NULL, /* 비밀번호 */
	DIVISION NVARCHAR2(5), /* 회원구분 */
	NICKNAME NVARCHAR2(8) NOT NULL /* 닉네임 */
)
	STORAGE (
		BUFFER_POOL DEFAULT
	)
	LOGGING
	NOCOMPRESS
	NOCACHE
	NOPARALLEL
	NOROWDEPENDENCIES
	DISABLE ROW MOVEMENT;

COMMENT ON COLUMN Member.PASSWORD IS '비밀번호';
